  jQuery.ev.loop($base+'/channel/curl_commands/stream/'+clientId, {
    Greeting: function(ev){
      jQuery('#messages').prepend('<li>'+ev.message+'</li>');
    },
    Color: function(ev){
      jQuery('body').css({ backgroundColor: ev.color });
    },
    '*': function(ev){
      try {
        jQuery('#messages').prepend('<li>'+ev.toSource()+'</li>');
      }
      catch(e) {
      }
    }
  });
